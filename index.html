<script>

    const getMoreIrregularOffset = () => (Math.random() - 0.5) * 80; 


    let islands = [

        { name: '제주도', offsetX: getMoreIrregularOffset(), offsetY: getMoreIrregularOffset() },

        { name: '독도', offsetX: getMoreIrregularOffset(), offsetY: getMoreIrregularOffset() },

        { name: '울릉도', offsetX: getMoreIrregularOffset(), offsetY: getMoreIrregularOffset() },

        { name: '거제도', offsetX: getMoreIrregularOffset(), offsetY: getMoreIrregularOffset() }

    ];

    

    let sunkenIslands = [];


    const state = {

        cw: { pos: 0, log: [], name: '술라바' },

        ccw: { pos: 0, log: [], name: '음왈리' }

    };


    let centerX, centerY, baseRadius;


    function updateDropdown() {

        const select = document.getElementById('island-to-sink');

        select.innerHTML = '';

        if (islands.length === 0) {

            select.innerHTML = '<option disabled selected>섬이 없습니다</option>';

            return;

        }

        islands.forEach((island, index) => {

            const opt = document.createElement('option');

            opt.value = index;

            opt.text = island.name;

            select.add(opt);

        });

    }


    function sinkSelectedIsland() {

        const select = document.getElementById('island-to-sink');

        if (islands.length === 0 || !select.value) return;

        

        const index = parseInt(select.value);

        const sunkIsland = islands.splice(index, 1)[0]; 

        

        const time = new Date().toLocaleTimeString('ko-KR', { hour: '2-digit', minute: '2-digit', second: '2-digit' });

        sunkenIslands.push(`${sunkIsland.name} (${time})`);

        

        ['cw', 'ccw'].forEach(key => {

            if (state[key].pos === index) {

                state[key].pos = 0; 

                if (islands.length > 0) {

                    state[key].log.push(`[🚨 긴급 대피] ➔ ${islands[0].name}`);

                }

            } else if (state[key].pos > index) {

                state[key].pos -= 1; 

            }

        });

        

        render();

        updateSinkModalContent();

    }


    function toggleSinkRecord() {

        const modal = document.getElementById('sink-modal');

        const overlay = document.getElementById('modal-overlay');

        const isShowing = modal.style.display === 'block';

        if (!isShowing) updateSinkModalContent(); 

        modal.style.display = isShowing ? 'none' : 'block';

        overlay.style.display = isShowing ? 'none' : 'block';

    }


    function updateSinkModalContent() {

        const list = document.getElementById('sink-list');

        if (sunkenIslands.length === 0) {

            list.innerHTML = "<li style='border:none; text-align:center;'>아직 바다 밑으로 가라앉은 섬이 없습니다.</li>";

        } else {

            list.innerHTML = sunkenIslands.map(record => `<li>${record}</li>`).join('');

        }

    }


    function updateName(key, newName) {

        state[key].name = newName || (key === 'cw' ? '술라바' : '음왈리');

        const el = document.getElementById(`item-${key}`);

        if(el) el.querySelector('.boat-name').innerText = state[key].name;

        

        const btn = document.getElementById(`btn-${key}-text`);

        const icon = key === 'cw' ? '🐚' : '🔲';

        const dirText = key === 'cw' ? ' (시계)' : ' (반시계)';

        if(btn) btn.innerText = `${icon} 출발${dirText}`;


        const logTitle = document.getElementById(`log-${key}-title`);

        if(logTitle) logTitle.innerText = state[key].name;

    }


    function updateLogPanel() {

        ['cw', 'ccw'].forEach(key => {

            const contentDiv = document.getElementById(`log-${key}-content`);

            if (state[key].log.length > 0) {

                contentDiv.innerHTML = state[key].log.join('<br>➔ '); 

            } else {

                contentDiv.innerHTML = '아직 출발하지 않았습니다.';

            }

            contentDiv.scrollTop = contentDiv.scrollHeight;

        });

    }


    function updateDimensions() {

        const paddingTop = 180;    

        const paddingBottom = 120; 

        const paddingSide = 200;   


        const safeWidth = window.innerWidth - (paddingSide * 2);

        const safeHeight = window.innerHeight - paddingTop - paddingBottom;


        centerX = paddingSide + (safeWidth / 2);

        centerY = paddingTop + (safeHeight / 2);


        baseRadius = Math.min(safeWidth, safeHeight) * 0.35; 

        if(baseRadius < 100) baseRadius = 100; 

    }


    function updatePaths() {

        const svg = document.getElementById('paths-svg');

        svg.innerHTML = ''; 

        if (islands.length < 2) return; 


        let pointsString = "";

        for (let i = 0; i < islands.length; i++) {

            const center = getIslandCenter(i);

            pointsString += `${center.x},${center.y} `;

        }

        const firstCenter = getIslandCenter(0);

        pointsString += `${firstCenter.x},${firstCenter.y}`;


        const polyline = document.createElementNS("http://www.w3.org/2000/svg", "polyline");

        polyline.setAttribute("points", pointsString);

        polyline.setAttribute("class", "route-line");

        svg.appendChild(polyline);

    }


    // --- 섬 가운데 글씨를 가리지 않도록 위치 계산 로직 변경 ---

    function getItemTargetPoint(key, index) {

        const center = getIslandCenter(index);

        // 술라바(cw)는 섬의 위쪽 테두리로, 음왈리(ccw)는 섬의 아래쪽 테두리로 고정 배치

        if (key === 'cw') {

            return { x: center.x, y: center.y - 45 }; // 섬 중앙에서 45px 위쪽

        } else {

            return { x: center.x, y: center.y + 45 }; // 섬 중앙에서 45px 아래쪽

        }

    }


    function render(movingItem = null) {

        updateDimensions(); 

        updateDropdown();


        const ocean = document.getElementById('ocean-map');

        

        islands.forEach((island, i) => {

            let imgId = `island-${i}`;

            let div = document.getElementById(imgId);

            if (!div) {

                div = document.createElement('div');

                div.id = imgId; div.className = 'island';

                ocean.appendChild(div);

            }

            div.innerText = island.name;

            const center = getIslandCenter(i);

            div.style.left = `${center.x - 42.5}px`;

            div.style.top = `${center.y - 42.5}px`;

        });


        const currentIslandDivs = document.querySelectorAll('.island');

        if (currentIslandDivs.length > islands.length) {

            for (let i = islands.length; i < currentIslandDivs.length; i++) {

                currentIslandDivs[i].remove();

            }

        }


        updatePaths();


        if (islands.length > 0) {

            renderItem('cw', 'item-cw', movingItem === 'cw');

            renderItem('ccw', 'item-ccw', movingItem === 'ccw');

        } else {

            document.querySelectorAll('.item').forEach(el => el.remove());

        }

        

        updateLogPanel();

    }


    function getIslandCenter(index) {

        if (islands.length === 0) return {x: centerX, y: centerY};

        const island = islands[index];

        const angle = (index / islands.length) * (2 * Math.PI) - (Math.PI / 2);

        const x = centerX + (baseRadius + island.offsetX) * Math.cos(angle);

        const y = centerY + (baseRadius + island.offsetY) * Math.sin(angle);

        return { x, y };

    }


    function renderItem(key, className, isMoving) {

        let elId = `item-${key}`;

        let div = document.getElementById(elId);

        if (!div) {

            div = document.createElement('div');

            div.id = elId;

            div.className = `item ${className}`;

            div.innerHTML = `<div class="boat-body"></div><span class="boat-name">${state[key].name}</span>`;

            document.getElementById('ocean-map').appendChild(div);

        } else {

            div.querySelector('.boat-name').innerText = state[key].name;

        }


        if (isMoving) return; 


        div.style.transition = 'left 1s ease-in-out, top 1s ease-in-out';

        

        const target = getItemTargetPoint(key, state[key].pos);

        div.style.left = `${target.x - 28}px`; 

        div.style.top = `${target.y - 20}px`; 

    }


    function startVoyage(direction) {

        if (islands.length < 2) return;

        const item = state[direction];

        const prevPos = item.pos;

        

        item.log.push(islands[prevPos].name);

        updateLogPanel();


        let nextPos;

        if (direction === 'cw') {

            nextPos = (prevPos + 1) % islands.length;

        } else {

            nextPos = (prevPos - 1 + islands.length) % islands.length;

        }

        item.pos = nextPos; 


        animateCanoe(direction, prevPos, nextPos);

    }


    function animateCanoe(key, startIdx, endIdx) {

        const canoe = document.getElementById(`item-${key}`);

        const startPoint = { x: parseFloat(canoe.style.left) + 28, y: parseFloat(canoe.style.top) + 20 };

        const endPoint = getItemTargetPoint(key, endIdx);

        

        const controlPoint = {

            x: (startPoint.x + endPoint.x) / 2 + (centerX - (startPoint.x + endPoint.x)/2) * 0.25,

            y: (startPoint.y + endPoint.y) / 2 + (centerY - (startPoint.y + endPoint.y)/2) * 0.25

        };


        const startTime = performance.now();

        const duration = 1400; 


        canoe.style.transition = 'none';

        const boatBody = canoe.querySelector('.boat-body');


        function step(currentTime) {

            const elapsed = currentTime - startTime;

            const progress = Math.min(elapsed / duration, 1);

            const p = easeInOutQuad(progress); 


            const currentX = (1 - p) * (1 - p) * startPoint.x + 2 * (1 - p) * p * controlPoint.x + p * p * endPoint.x;

            const currentY = (1 - p) * (1 - p) * startPoint.y + 2 * (1 - p) * p * controlPoint.y + p * p * endPoint.y;


            const nextP = Math.min(progress + 0.01, 1);

            const nextX = (1 - nextP) * (1 - nextP) * startPoint.x + 2 * (1 - nextP) * nextP * controlPoint.x + nextP * nextP * endPoint.x;

            const nextY = (1 - nextP) * (1 - nextP) * startPoint.y + 2 * (1 - nextP) * nextP * controlPoint.y + nextP * nextP * endPoint.y;

            const angle = Math.atan2(nextY - currentY, nextX - currentX) * 180 / Math.PI;


            canoe.style.left = `${currentX - 28}px`;

            canoe.style.top = `${currentY - 20}px`;

            boatBody.style.transform = `rotate(${angle}deg)`; 


            if (progress < 1) requestAnimationFrame(step);

            else render();

        }


        render(key); 

        requestAnimationFrame(step);

    }


    function easeInOutQuad(t) { return t < 0.5 ? 2 * t * t : -1 + (4 - 2 * t) * t; }


    function addIsland() {

        const input = document.getElementById('island-name');

        const name = input.value || `새로운 섬 ${islands.length + 1}`;

        islands.push({ name: name, offsetX: getMoreIrregularOffset(), offsetY: getMoreIrregularOffset() });

        input.value = '';

        render();

    }


    render();

    window.addEventListener('resize', () => render());
